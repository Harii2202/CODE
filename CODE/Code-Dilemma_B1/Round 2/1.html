2) C: Find Maximum in an Array

#include <stdio.h>
int main() {
    int arr[] = {3, 5, 7, 2, 8};
    int max = arr[0];
    for (int i = 1; i <= 5; i++) { 
        if (arr[i] > max) {
            max = arr[i];
        }
    }
    printf("Maximum value is %d", max);
    return 0;
}

9) C: Fibonacci Series

    #include <stdio.h>
    int main() {
        int n = 5, t1 = 0, t2 = 1, nextTerm;
        printf("Fibonacci Series: %d, %d, ", t1, t2);
        for (int i = 1; i <= n-1; ++i) { 
            nextTerm = t1 + t2;
            printf("%d, ", nextTerm);
            t1 = t2;
            t2 = nextTerm;
        }
        return 0;
    }

13) C: Find GCD

    #include <stdio.h>
    int main() {
        int a = 36, b = 60, gcd = 1;
        for (int i = 1; i < a || i < b; ++i) {
            if (a % i == 0 && b % i == 0)
                gcd = i;
        }
        printf("GCD is %d", gcd);
        return 1;
    }
